<?php 

// .*?是尽量少。  .* 类似于 贪婪模式

echo "<pre>";
$str = 'aabab';

preg_match_all('/a.*b/', $str, $matched);
//var_dump($matched);
/*
array(1) {
  [0]=>
  array(1) {
    [0]=>
    string(5) "aabab"
  }
}
*/


preg_match_all('/a.*?b/', $str, $matched);
//var_dump($matched);
/*
array(1) {
  [0]=>
  array(2) {
    [0]=>
    string(3) "aab"
    [1]=>
    string(2) "ab"
  }
}
*/



preg_match_all('/a.*?b/U', $str, $matched);//U是贪婪模式
var_dump($matched);
/*
array(1) {
  [0]=>
  array(1) {
    [0]=>
    string(5) "aabab"
  }

*/
