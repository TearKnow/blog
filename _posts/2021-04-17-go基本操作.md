```
func swap(x, y string){

    tmp := x

    x = y

    y = tmp

    var a = "aa"

    var b = "bb"

    swap(a, b)

    fmt.Println(a, b)
```

```
func swap(x *string, y *string){

    tmp := *x

    *x = *y

    *y = tmp

}

func main() {

    var a = "aa"

    var b = "bb"

    swap(&a, &b)

    fmt.Println(a, b)

}
```
输出 bb aa

#### 19.2 通道 channel，计算就和
```
package main

import (

    "fmt"

)

func sum(s []int, c chan int){

    sum :=0

    for _,v :=range s{

        sum += v

    }

    c <- sum  //把sum发送到c通道

}

func main(){

    s := []int{7, 2, 8, -9, 4, 2}

    c := make(chan int)

    go sum(s[:len(s) / 2], c)

    go sum(s[len(s) / 2:], c)

    x, y := <-c, <-c //从通道c中接收值

    fmt.Println(x, y , x+y)  //输出：-3 17 14

}
```
