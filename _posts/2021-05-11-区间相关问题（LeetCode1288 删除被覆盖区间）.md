区间相关的问题，步骤：
1. 排序
2. 动手画图

```php
class Solution {

    /**
     * @param Integer[][] $intervals
     * @return Integer
	 //起点升序，终点降序 排列有3中情况
     1. 
        -------
          ----
     2. -------
          -------
     3. -------
                   -----
	 分别是覆盖，相交，无交集
     */
    function removeCoveredIntervals($intervals) {
        usort($intervals, function($a, $b){
            if($a[0] < $b[0]){
                return -1;
            }elseif($a[0] > $b[0]){
                return 1;
            }else{//相等的情况，终点降序
                if($a[1] > $b[1]){
                    return -1;
                }else{
                    return 1;
                }
            }
        });
        
        $left = $intervals[0][0];
        $right = $intervals[0][1];
        
        $rev = 0;//有x个被覆盖了
        for($i = 1; $i <= count($intervals) - 1; $i++){
            //1. 覆盖 
            if($intervals[$i][0] >= $left
              && $intervals[$i][1] <= $right
            ){
                $rev++;
            }
            
            //2. 相交 
            if($intervals[$i][0] >= $left
              && $intervals[$i][1] >= $right)
            {
                $right = $intervals[$i][1];
            }
            
            
            //3. 无交集//重新更新left 和 right
            if($intervals[$i][0] > $right){
                $left = $intervals[$i][0];
                $right = $intervals[$i][1];
            }
            
        }
        
        return count($intervals) - $rev;
    }
}
```
