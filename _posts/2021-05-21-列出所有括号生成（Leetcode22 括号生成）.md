Leetcode22. 括号生成

### 方法一. 回溯法列出所有可能，在判断是否是有效的
```php
<?php 
class Solution {

    /**
     * @param Integer $n
     * @return String[]
     */
    public $maybe = [];
    function generateParenthesis($n) {
        $this->son($n, []);
        $show = [];
        foreach($this->maybe as $m){
            $result = $this->valid($m);
            if($result){
                $show[] = $m;
            }
        };
        return $show;
    }
    
    public function valid($str){
        $arr = [];
        while($str){
            $first = substr($str, 0, 1);
            $str = substr($str, 1);//排除第一个后，剩余的
            if($first == '('){
                $arr[] = $first;
            }elseif($first == ')'){//需要去pop
                if(!$arr){
                    return false;
                }
                if(end($arr) == '('){
                    array_pop($arr);
                }
            }
        }
        return !$arr ? true : false;
    }
    
    public function son($n, $selected){
        if(count($selected) >= $n * 2){
            if(end($selected) == '('){
                
            }else{
                $this->maybe[] = implode('', $selected);
            }
            
            return;
        }
        $map = ['(', ')'];
        for($i = 0; $i <= 1; $i++){
            $tmpCount = [];
            foreach($selected as $s){
                @$tmpCount[$s] += 1;
            }
            if(@$selected[0] == ')'){//第一个不可能)开始
                continue;
            }
            
            //减枝 todo ，根据前面几个就判断是否后面的可能
            
            if(@$tmpCount[$map[$i]] >= $n){//减枝
                continue;
            }
            
            $selected[] = $map[$i];
            
            $this->son($n, $selected);
            
            array_pop($selected);
        }
    }
}

$obj = new Solution;
$x = $obj->generateParenthesis(8);
var_dump($x);
```
